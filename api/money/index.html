@w5s/money

W5S Money module (@w5s/money)

Installation​

npm install @w5s/money

Usage​

import { Money, EUR } from '@w5s/money';

export function main(): void {
  const price = EUR(10);
  const discount = EUR(2);
  const result = Money['-'](price, discount);
  console.log(result); // > Result.Ok(EUR(8))
}
