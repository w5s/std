Headers

  • Headers(values: Iterable<readonly [string, string]> | Record<string, string>): Headers

  • HTTP header record constructor

    @example
    const headersFromIterable = Headers([
     ['key1', 'value1'],
     ['key2', 'value2']
    ]);// { key1: 'value1, key2: 'value2' }
    const headersFromObject = Headers({
     key1: 'value1',
     key2: 'value2'
    });// { key1: 'value1, key2: 'value2' }
